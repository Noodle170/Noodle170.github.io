Sprawdzenie czy liczba jest parzysta

function parzysta_zad() {
var x = parseInt(document.getElementById("parzysta").value);
if (x % 2 == 0) {
document.getElementById("wynik1").innerHTML = "Liczba " + x + " jest parzysta.";
} else {
document.getElementById("wynik1").innerHTML = "Liczba " + x + " nie jest parzysta.";
}
if (x == 0) {
document.getElementById("wynik1").innerHTML = "Podałeś " + x + ".";
}
}

}

Silnia

function silnia_zad() {
var x = parseInt(document.getElementById("silnia").value);
var z = 1;
for (var i = 1; i <= x; i++) {
z *= i;
}
document.getElementById("wynik4").innerHTML = "Silnia z " + x + " wynosi: " + z;
}
